function findRecipe(){

}

else if(input.includes("tomato") && input.includes("pasta")){
result="🍝 Simple Tomato Pasta: Boil pasta, cook tomatoes with olive oil, garlic, salt and herbs. Mix together.";
}

else if((input.includes("egg") || input.includes("eggs")) && input.includes("vegetable")){

}

else if(input.includes("beans") && input.includes("rice")){
result="🍛 Beans & Rice (Rajma Chawal): Cook beans with onion, garlic, tomato and spices. Serve with steamed rice.";
}

else if(input.includes("potato") && input.includes("salt") && input.includes("chilli")){
result="🥔 Salty Boiled Potatoes with Chilli Pieces: Boil potatoes, sprinkle salt and chopped chilli, drizzle with butter.";
}

else if(input.includes("bread") && input.includes("egg")){
result="🍞 Egg Toast: Dip bread in beaten egg and fry in a pan until golden.";
}

else if(input.includes("rice") && input.includes("egg")){
result="🍳 Egg Fried Rice: Fry rice with scrambled eggs, soy sauce, and vegetables.";
}

else if(input.includes("milk") && input.includes("banana")){
result="🥤 Banana Milk Smoothie: Blend banana, milk, honey and ice.";
}

else if(input.includes("tomato") && input.includes("egg")){
result="🍅 Tomato Egg Scramble: Cook tomatoes in a pan then scramble eggs into them.";
}

else if(input.includes("cheese") && input.includes("bread")){
result="🧀 Grilled Cheese Sandwich: Butter bread, add cheese and grill until crispy.";
}

else{
result="🤔 Try simple dishes like fried rice, omelette, vegetable soup, grilled sandwich, or pasta based on your ingredients.";
}

document.getElementById("result").innerHTML=result;

}
